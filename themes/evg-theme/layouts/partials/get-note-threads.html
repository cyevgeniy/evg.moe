{{- $notes := where .Site.RegularPages.ByPublishDate.Reverse "Section" "notes" -}} {{- $threads := slice -}} {{- $seen := dict -}} {{- range $notes -}} {{/* Use the file path as the fallback key so standalone notes still behave like one-item threads. */}} {{- $threadKey := .File.Path -}} {{- with .Params.thread -}} {{/* Trim front matter values to avoid accidental tabs/spaces splitting one logical thread into two. */}} {{- $threadKey = strings.TrimSpace (printf "%v" .) -}} {{- end -}} {{- if not (index $seen $threadKey) -}} {{- $threadNotes := slice . -}} {{- if .Params.thread -}} {{/* Hugo cannot efficiently "where" on a normalized value, so we build the matching slice manually. */}} {{- $threadNotes = slice -}} {{- range $notes -}} {{- $candidateThreadKey := .File.Path -}} {{- with .Params.thread -}} {{- $candidateThreadKey = strings.TrimSpace (printf "%v" .) -}} {{- end -}} {{- if eq $candidateThreadKey $threadKey -}} {{- $threadNotes = $threadNotes | append . -}} {{- end -}} {{- end -}} {{/* File names like 2025-06-14-aa.md keep replies in a predictable order inside the thread. */}} {{- $threadNotes = sort $threadNotes "File.Path" "asc" -}} {{- end -}} {{- $head := index $threadNotes 0 -}} {{- $last := index $threadNotes (sub (len $threadNotes) 1) -}} {{- $threads = $threads | append (dict "key" $threadKey "notes" $threadNotes "head" $head "last" $last ) -}} {{- $seen = merge $seen (dict $threadKey true) -}} {{- end -}} {{- end -}} {{- return $threads -}}
